### 3. 智能文案生成器
```javascript
// lib/copywriter.js
class MysticalCopywriter {
  // 根据配对等级生成不同风格的文案
  generatePoem(grade, userMBTI, lobsterMBTI) {
    const poems = {
      S: this._generateSPoem(),
      A: this._generateAPoem(),
      B: this._generateBPoem(),
      C: this._generateCPoem(),
      D: this._generateDPoem()
    };
    return poems[grade];
  }

  _generateSPoem() {
    const templates = [
      "火与火的相遇，星光璀璨，思维在跳舞，灵感在狂欢",
      "金色的缘分线，将你们紧紧相连，天生的拍档，共绘未来画卷",
      "星辰排列成行，只为见证你们的相遇，命中注定的默契"
    ];
    return templates[Math.floor(Math.random() * templates.length)];
  }

  // 根据MBTI组合生成个性化洞察
  generateInsight(userMBTI, lobsterMBTI) {
    const combinations = {
      "ENFP-ENTP": {
        resonance: "完美火象共振",
        strength: "创意无限，火花四溅",
        weakness: "执行力不足，细节容易遗漏",
        advice: "设定明确的Deadline，建立检查点机制"
      },
      // ... 其他组合
    };
    return combinations[`${userMBTI}-${lobsterMBTI}`] || this._getDefaultInsight();
  }
}
```
